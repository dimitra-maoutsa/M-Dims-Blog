@jax.jit
def grad_vf(t,y,state):
  s = mutils.get_model_fn(model_s, 
                          state.params_ema if config.eval.use_ema else state.model_params, 
                          train=False)
  dsdx = jax.grad(lambda _t, _x: s(_t*jnp.ones((_x.shape[0],1)), _x).sum(), argnums=1)
  return dsdx(t,y)
